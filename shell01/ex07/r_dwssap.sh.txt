cat /etc/passwd | grep -v '^#' | sed -n 'p;n' | cut -d':' -f1 | rev | sort -r | awk 'NR>=7 && NR<=15' | paste -sd ',' - | sed 's/,/, /g' | sed 's/$/./'

grep -v '^#' : Supprime les commentaires

sed -n 'p;n' : Affiche toutes les autres lignes, en commençant par la première. L'option -n demande à sed de ne pas imprimer automatiquement chaque ligne, et p;n signifie "imprimer la ligne actuelle et sauter la suivante".

cut -d':' -f1 : Sélectionne seulement les noms

sort -r : sort desc

sed 's/,/, /g' : Ajoute un espace après chaque virgule.

sed 's/$/./' : Ajoute un point à la fin de la ligne.
