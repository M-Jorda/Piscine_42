#include <stdio.h>

void ft_print_comb(void) {
    int a = 0;
    int b;
    int c;

    while (a <= 7) {
        b = a + 1;
        while (b <= 8) {
            c = b + 1;
            while (c <= 9) {
                putchar(a + '0');
                putchar(b + '0');
                putchar(c + '0');
                if (!(a == 7 && b == 8 && c == 9)) {
                    putchar(',');
                    putchar(' ');
                }
                c++;
            }
            b++;
        }
        a++;
    }
}

int main(void) {
    ft_print_comb();
    return 0;
}
