#include <stdio.h>

void print_combination(int n1, int n2) {
    putchar((n1 / 10) + '0');
    putchar((n1 % 10) + '0');
    putchar(' ');
    putchar((n2 / 10) + '0');
    putchar((n2 % 10) + '0');
    if (n1 != 98 || n2 != 99) {
        putchar(',');
        putchar(' ');
    }
}

void ft_print_comb2(void) {
    int n1 = 0, n2;
    while (n1 <= 98) {
        n2 = n1 + 1;
        while (n2 <= 99) {
            print_combination(n1, n2);
            n2++;
        }
        n1++;
    }
    putchar('\n');
}

int main() {
    ft_print_comb2();
    return 0;
}
