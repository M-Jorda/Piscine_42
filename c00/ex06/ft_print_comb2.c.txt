#include <stdio.h>

void ft_print_comb2(void) {
    int i = 0;
    int j;

    while (i <= 98) {
        j = i + 1;
        while (j <= 99) {
            putchar((i / 10) + '0');
            putchar((i % 10) + '0');
            putchar(' ');
            putchar((j / 10) + '0');
            putchar((j % 10) + '0');
            if (!(i == 98 && j == 99)) {
                putchar(',');
                putchar(' ');
            }
            j++;
        }
        i++;
    }
}

int main(void) {
    ft_print_comb2();
    return 0;
}
