#include <stdio.h>

void ft_print_combn_util(int *arr, int n, int index, int start) {
    if (index == n) {
        for (int i = 0; i < n; i++) {
            putchar(arr[i] + '0');
        }
        if (arr[0] < 10 - n) {
            putchar(',');
            putchar(' ');
        }
        return;
    }

    for (int i = start; i <= 9; i++) {
        arr[index] = i;
        ft_print_combn_util(arr, n, index + 1, i + 1);
    }
}

void ft_print_combn(int n) {
    if (n <= 0 || n >= 10) return;

    int arr[10] = {0};
    ft_print_combn_util(arr, n, 0, 0);
}

int main(void) {
    ft_print_combn(2);
    putchar('\n');

    return 0;
}
